012/*
013 BlackRock cipher
014
015 (h/t Marsh Ray @marshray for this idea)
016
017 This is a randomization/reshuffling function based on a crypto
018 "Feistel network" as described in the paper:
019
020 'Ciphers with Arbitrary Finite Domains' 
021 by John Black and Phillip Rogaway 
022 http://www.cs.ucdavis.edu/~rogaway/papers/subset.pdf
023
024 This is a crypto-like construction that encrypts an arbitrary sized
025 range. Given a number in the range [0..9999], it'll produce a mapping 
026 to a distinct different number in the same range (and back again).
027 In other words, it randomizes the order of numbers in a sequence.
028
029 For example, it can be used to  randomize the sequence [0..9]:
030    
031 0 ->      6
032 1 ->      4
033 2 ->      8
034 3 ->      1
035 4 ->      9
036 5 ->      3
037 6 ->      0
038 7 ->      5
039 8 ->      2
040 9 ->      7
041
042 As you can see on the right hand side, the numbers are in random
043 order, and they don't repeat.
044
045 This is created for resource allocation. We can take an index variable
046 and increment it during a scan, then use this function to
047 randomize it, yet be assured that we've allocated every resource
048 within the range.
049
050 The cryptographic strength of this construction depends upon the 
051 number of rounds, and the exact nature of the inner "F()" function.
052 Because it's a Feistal network, that "F()" function can be almost
053 anything.
054
055 We don't care about cryptographic strength, just speed, so we are
056 using a trivial F() function.
057
058 This is a class of "format-preserving encryption". There are 
059 probably better constructions than what I'm using.
060 */

112    private static class Parameters {
113
114        private final long a;
115        private final long b;

118            this.a = a;

128    public PermutationGenerator(@Nonnegative long range, long seed) {
129        this.range = range;
130        this.seed = seed;
131        Parameters parameters = init(range);
132        this.a = parameters.a;
133        this.b = parameters.b;
134    }

136    @Nonnegative

138        return range;

141    /***************************************************************************
142     ***************************************************************************/
143    private static Parameters init(final long range) {
144        /* This algorithm gets very non-random at small numbers, so I'm going
145         * to try to fix some constants here to make it work. It doesn't have
146         * to be good, since it's kinda pointless having ranges this small */
147        switch ((int) range) {
148            case 0:
149                return new Parameters(0, 0);
150            case 1:
151                return new Parameters(1, 1);
152            case 2:
153                return new Parameters(1, 2);
154            case 3:
155                return new Parameters(2, 2);
156            case 4:
157            case 5:
158            case 6:
159                return new Parameters(2, 3);
160            case 7:
161            case 8:
162                return new Parameters(3, 3);
163        }
164
165        double foo = Math.sqrt((double) range);
166        long a = (long) (foo - 1);
167        long b = (long) (foo + 1);
168        while (a * b <= range)
169            b++;
170        return new Parameters(a, b);
171    }
172    /***************************************************************************
173     * This is a random meaningless function. Well, if we actually wanted
174     * crypto-strength, we'd have to improve it, but for now, we just want
175     * some random properties.
176     ***************************************************************************/
177    private static long[] primes = new long[]{961752031L, 982324657L, 15485843L, 961752031L,};
178
179    @Nonnegative
180    private static long F(@Nonnegative final int round, @Nonnegative final long R, final long seed) {
181        long out = ((R << (R & 0x4)) + R + seed);
182        out ^= sbox[(int) (out & 0xF)];
183        out = (((primes[round] * out + 25L) ^ out) + round);
184        // LOG.info("F(" + round + ", " + R + ", " + seed + " = " + out);
185        return out;
186    }
187
188    @Nonnegative
189    private static long fe(@Nonnegative int rounds, @Nonnegative long a, @Nonnegative long b, @Nonnegative long m, long seed) {
190        long L = m % a;
191        long R = m / a;
192
193        for (int round = 1; round <= rounds; round++) {
194            // LOG.info("round=" + round + ", L=" + L + ", R=" + R);
195            long tmp;
196            if ((round & 1) != 0) {
197                tmp = LongMath.mod(L + F(round, R, seed), a);
198            } else {
199                tmp = LongMath.mod(L + F(round, R, seed), b);
200            }
201            L = R;
202            R = tmp;
203            assert L >= 0 && R >= 0 : "a=" + a + ", b=" + b + ", m=" + m + ", round=" + round + ", L=" + L + ", R=" + R;
204        }
205
206        if ((rounds & 1) != 0) {
207            return a * L + R;
208        } else {
209            return a * R + L;
210        }
211    }
212
213    @Nonnegative
214    private static long unfe(@Nonnegative int rounds, @Nonnegative long a, @Nonnegative long b, @Nonnegative long m, long seed) {
215        long L, R;
216
217        if ((rounds & 1) != 0) {
218            R = m % a;
219            L = m / a;
220        } else {
221            L = m % a;
222            R = m / a;
223        }
224
225        for (int round = rounds; round >= 1; round--) {
226            long tmp;
227            if ((round & 1) != 0) {
228                tmp = F(round, L, seed);
229                tmp = LongMath.mod(R - tmp, a);
230            } else {
231                tmp = F(round, L, seed);
232                tmp = LongMath.mod(R - tmp, b);
233            }
234            R = L;
235            L = tmp;
236        }
237        return a * R + L;
238    }
239
240    private void check(@Nonnegative long v, @Nonnull String purpose) {
241        if (v < 0)
242            throw new IllegalArgumentException("Illegal negative " + purpose + " " + v);
243        if (v >= range)
244            throw new IllegalArgumentException("Illegal " + purpose + " " + v + " >= " + range);
245    }
246
247    @Nonnegative
248    public long shuffle(@Nonnegative long m) {
249        check(m, "input");
250        long c = fe(rounds, a, b, m, seed);
251        while (c >= range)
252            c = fe(rounds, a, b, c, seed);
253        if (c < 0)
254            throw new IllegalStateException(this + " generated " + c + " for input " + m);
255        return c;
256    }
257
258    @Nonnegative
259    public long unshuffle(@Nonnegative long m) {
260        check(m, "input");
261        long c = unfe(rounds, a, b, m, seed);
262        while (c >= range)
263            c = unfe(rounds, a, b, c, seed);
264        if (c < 0)
265            throw new IllegalStateException(this + " generated " + c + " for input " + m);
266        return c;
267    }

269    @Override
270    public String toString() {
271        return Objects.toStringHelper(this)
272                .add("range", range)
273                .add("seed", seed)
274                .add("rounds", rounds)
275                .add("a", a)
276                .add("b", b)
277                .toString();
278    }
