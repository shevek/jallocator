005package org.anarres.jallocator;
006
007import com.google.common.base.Objects;
008import com.google.common.base.Preconditions;
009import java.util.Iterator;
010import java.util.Random;
011import java.util.concurrent.atomic.AtomicLong;
012import javax.annotation.Nonnull;
013import javax.annotation.concurrent.ThreadSafe;
014
015/**
016 *
017 * @author shevek
018 */
019@ThreadSafe
020public class ResourceAllocator<T> implements Iterable<T> {
021
022    // private static final Logger LOG = LoggerFactory.getLogger(ResourceAllocator.class);
023    private static Random RANDOM = new Random();
024    @Nonnull
025    private ResourceProvider<T> resourceProvider;
026    @Nonnull
027    private PermutationGenerator generator;
028    @Nonnull
029    private final AtomicLong counter = new AtomicLong(0);
030
031    public ResourceAllocator(@Nonnull ResourceProvider<T> resourceProvider) {
032        setResourceProvider(resourceProvider);
033    }
034
035    @Nonnull
036    public ResourceProvider<T> getResourceProvider() {
037        return resourceProvider;
038    }
039
040    // TODO: I wish this were atomic. It'll violate the contract of ResourceProvider and presumably OOB otherwise.
041    public void setResourceProvider(@Nonnull ResourceProvider<T> resourceProvider) {
042        if (!Objects.equal(this.resourceProvider, resourceProvider)) {
043            this.resourceProvider = Preconditions.checkNotNull(resourceProvider, "ResourceProvider was null.");
044            this.generator = new PermutationGenerator(resourceProvider.getResourceCount(), System.currentTimeMillis() ^ RANDOM.nextLong());
045        }
046    }
047
048    @Nonnull
049    @Override
050    public Iterator<T> iterator() {
051        return new ResourceIterator<T>(resourceProvider, generator, counter);
052    }
053}
