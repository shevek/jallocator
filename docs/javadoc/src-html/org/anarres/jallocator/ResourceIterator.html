005package org.anarres.jallocator;
006
007import com.google.common.collect.AbstractIterator;
008import com.google.common.math.LongMath;
009import java.util.concurrent.atomic.AtomicLong;
010import javax.annotation.Nonnegative;
011import javax.annotation.Nonnull;
012
013/**
014 *
015 * @author shevek
016 */
017public class ResourceIterator<T> extends AbstractIterator<T> {
018
019    // private static final Logger LOG = LoggerFactory.getLogger(ResourceIterator.class);
020    @Nonnull
021    private final ResourceProvider<T> resourceProvider;
022    @Nonnull
023    private final PermutationGenerator generator;
024    @Nonnull
025    private final AtomicLong counter;
026    @Nonnegative
027    private final long counterStart;
028    @Nonnegative
029    private final long counterRange;
030
031    public ResourceIterator(@Nonnull ResourceProvider<T> resourceProvider, @Nonnull PermutationGenerator generator, @Nonnull AtomicLong counter) {
032        this.resourceProvider = resourceProvider;
033        this.generator = generator;
034        this.counter = counter;
035        this.counterStart = counter.get();
036        this.counterRange = resourceProvider.getResourceCount();
037    }
038
039    // Watch carefully...
040    @Override
041    protected T computeNext() {
042        // TODO: This has a weakness that if it overruns, it will consume one more value
043        // from the counter than it attempts to allocate. This could potentially prevent
044        // another client from seeing (and allocating) that value, although if
045        // the algorithm loops around the permutation faster than the expiry time
046        // of resources, then the resource is known unallocated, as were it
047        // allocated, we would have allocated it on the first pass.
048        long counterValue = counter.getAndIncrement();
049        // LOG.info("Start=" + counterStart + ", value=" + counterValue + ", range=" + counterRange);
050        if (counterValue >= counterStart + counterRange)    // overflow will fuck us.
051            return endOfData();
052        long sequenceValue = LongMath.mod(counterValue, counterRange);
053        long randomValue = generator.shuffle(sequenceValue);
054        return resourceProvider.getResource(randomValue);
055    }
056}
